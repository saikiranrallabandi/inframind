Quick Start

from transformers import AutoModelForCausalLM, AutoTokenizer # Load model model = AutoModelForCausalLM.from_pretrained("srallabandi0225/inframind-0.5b-grpo") tokenizer = AutoTokenizer.from_pretrained("srallabandi0225/inframind-0.5b-grpo") # Generate Terraform prompt = """### Instruction: Create Terraform for AWS EC2 instance ### Response: """ inputs = tokenizer(prompt, return_tensors="pt") outputs = model.generate(**inputs, max_new_tokens=512) print(tokenizer.decode(outputs[0], skip_special_tokens=True))
